<script type="text/javascript">

    var _paginationCurrentPageStudentGrid = 1;
    var _paginationTotalPageStudentGrid = 0;

    if ('StudentGrid' != "") {
        function PagerDataBoundEmployeeGrid(_currentPage, _recordPerPage, _totalRecords) {
            _paginationCurrentPageEmployeeGrid = _currentPage;
            _paginationTotalPageEmployeeGrid = CountPages(_totalRecords, _recordPerPage)
            //FormatPagination()
            StudentGridFormatPagination();
        }
    }

    function CheckIsButtonEnabled(itm, e){
        var _parentLi = $(itm).parent()
        return _parentLi.hasClass('disabled')
    }

    function StudentGridFirstButtonClick(itm, e) {
        var IsEnabled = CheckIsButtonEnabled(itm, e)
        if (IsEnabled == false) {
            _paginationCurrentPageStudentGrid = 1;
            EmployeeGridPageChange(_paginationCurrentPageEmployeeGrid)
        }
    }

    function StudentGridPreviousButtonClick(itm, e) {
        var IsEnabled = CheckIsButtonEnabled(itm, e)
        if (IsEnabled == false) {
            _paginationCurrentPageStudentGrid -=1;
            StudentGridPageChange(_paginationCurrentPageStudentGrid)
        }
    }

    function StudentGridNextButtonClick(itm, e) {
        var IsEnabled = CheckIsButtonEnabled(itm, e)
        if (IsEnabled == false) {
            _paginationCurrentPageEmployeeGrid +=1;
            StudentGridPageChange(_paginationCurrentPageStudentGrid)
        }
    }

    function StudentGridLastButtonClick(itm, e) {
        var IsEnabled = CheckIsButtonEnabled(itm, e)
        if (IsEnabled == false) {
            _paginationCurrentPageStudentGrid = _paginationTotalPageStudentGrid;
            StudentGridPageChange(_paginationCurrentPageStudentGrid)
        }
    }

    function CountPages(totalRecords, recordsPerPage){
        if (totalRecords == 0)
        {
            return 1;
        }
        else{
            return Math.ceil(totalRecords / recordsPerPage);
        }
    }

    function StudentGridFormatPagination(){

        $('#txtPageNumber_StudentGrid').val('')
        $('#txtPageNumber_StudentGrid').val(_paginationCurrentPageStudentGrid)
        $('#spanTotalPages_StudentGrid').html(Math.floor(_paginationTotalPageStudentGrid))

        //for first button
        if (_paginationCurrentPageStudentGrid == 1) {
            $('#liFirst_StudentGrid').addClass('disabled')
        }
        else
        {
            $('#liFirst_StudentGrid').removeClass('disabled')
        }

        //for Previous button
        if (_paginationCurrentPageStudentGrid <= 1) {
            $('#liPrevious_StudentGrid').addClass('disabled')
        }
        else{
            $('#liPrevious_StudentGrid').removeClass('disabled')
        }

        //for last button
        if (_paginationCurrentPageStudentGrid == _paginationTotalPageStudentGrid) {
            $('#liLast_StudentGrid').addClass('disabled')
            $('#liNext_StudentGrid').addClass('disabled')
        }
        else
        {
            $('#liLast_StudentGrid').removeClass('disabled')
            $('#liNext_StudentGrid').removeClass('disabled')
        }
    }

    function CheckKey(itm, e){
        //enter click
        if ($.inArray(e.keyCode, [46, 8, 9, 27, 110, 190]) !== -1 ||
            // Allow: Ctrl+A, Command+A
           (e.keyCode === 65 && (e.ctrlKey === true || e.metaKey === true)) ||
            // Allow: home, end, left, right, down, up
           (e.keyCode >= 35 && e.keyCode <= 40)) {
            // let it happen, don't do anything
            return;
        }
        // Ensure that it is a number and stop the keypress
        if ((e.shiftKey || (e.keyCode < 48 || e.keyCode > 57)) && (e.keyCode < 96 || e.keyCode > 105)) {
            e.preventDefault();
        }

        if (e.keyCode == 13) {
            //return PaginationGoToSpecificPage(itm, e)
            return StudentGridPaginationGoToSpecificPage(itm, e);
        }
    }

    function StudentGridPaginationGoToSpecificPage(itm, e){
        var enteredNum = Number($(itm).val());
        if (enteredNum != undefined && enteredNum != "" && enteredNum != NaN && _paginationCurrentPageEmployeeGrid != enteredNum) {
            //AutoSetPageNumber(enteredNum)
            StudentGridAutoSetPageNumber(enteredNum);
            StudentGridPageChange(_paginationCurrentPageStudentGrid)
            return true;
        }
        return false;
    }

    function StudentGridAutoSetPageNumber(enteredNum){
        if (enteredNum <= 0) {
            $('#txtPageNumber_StudentGrid').val(1)
            _paginationCurrentPageStudentGrid = 1;
        }
        else if (enteredNum > _paginationTotalPageStudentGrid) {
            $('#txtPageNumber_StudentGrid').val(_paginationTotalPageStudentGrid)
            _paginationCurrentPageStudentGrid = _paginationTotalPageStudentGrid;
        }
        else{
            _paginationCurrentPageStudentGrid = enteredNum;
        }
    }

<script type="text/javascript">
    var currentPage = 1;
    var grdStudent;

    $(document).ready(function () {
        setTimeout(InitializeStudentGrid, 200);

        $("input[type='text']").keyup(function (event) {
            if (event.keyCode === 13) {
                $("#btnSearch").click();
            }
        });
    });


    function SearchStudentParameters(d) {
        var name = $('#Name').val();        
        var mobileNo = $('#MobileNo').val();

        d.PageNum = currentPage;
        d.Name = name;
        d.MobileNo = mobileNo;
    }

    function StudentGridDataBound(response) {
        if (response.IsSuccess == true) {
            PagerDataBoundStudentGrid(response.CurrentPage, response.PageSize, response.TotalRecords);
            return response.data;
        }
        else {
            showToaster(response.ErrorMsg, ToasterType.ERROR);
        }
    }


    function ReloadStudentGrid() {
        grdEmployee.ajax.url("/Student/Search").load();
    }

    function StudentGridPageChange(cPage) {
        currentPage = cPage;
        ReloadStudentGrid();
    }


    function InitializeStudentGrid() {
        grdStudent = $('#tblStudent').DataTable({
            serverSide: true,
            ajax: {
                beforeSend: function (xhr) {
                    showLoader();
                },
                error: function (jqXHR, textStatus, errorThrown) {
                    ShowErrorMessage(jqXHR.status, errorThrown);
                },
                complete: function () {
                    hideLoader();
                },
                url: '/Student/Search',
                data: SearchStudentParameters,
                dataSrc: StudentGridDataBound,
                type: "POST"
            },
            scrollCollapse: true,
            serverSide: true,
            processing: true,
            processing: "<span class='glyphicon glyphicon-refresh glyphicon-refresh-animate'>",
            ordering: true,
            bDestroy: true,
            paging: false,
            searching: false,
            bLengthChange: false,
            info: false,
            columnDefs: [
            {
                targets: 0,
                data: "CreatedOn",
                orderable: true,
                "visible": false
            },
            {
                targets: 1,
                data: "FirstName",
                orderable: true
            },
            {
                targets: 2,
                data: "LastName",
                orderable: true
            },
            {
                targets: 3,
                data: "MobileNo",
                orderable: true
            },
            {
                targets: 4,
                data: "Skills",
                orderable: true
            },
            {
                targets: 5,
                data: "Id",
                orderable: false,
                className: "text-center",                
                render: function (data, type, full, meta) {
                    return '<a href="javascript:void(0);" onclick="GotoEmployeeEdit(\'' + full.Id + '\')" '
                        + ' class="btn btn-success btn-xs" title="Edit">Edit</a>&nbsp;&nbsp;'
                        + '<a href="javascript:void(0);" onclick="deleteEmployee(this,event,\'' + full.Id + '\')" '
                        + ' class="btn btn-danger btn-xs" title="Delete">Delete</a>';
                }
            }
            ],
            order: [[0, "asc"]],
            fnCreatedRow: function (nRow, aData, iDataIndex) {
                $(nRow).attr('ID', aData.ID)
            }
        })
    }

    function GotoStudentEdit(Id) {
        var url = '/Student/Edit/' + Id;
        redirectTo(url);
    }

    function searchStudent() {
        currentPage = 1;
        ReloadStudentGrid();
    }

    function clearStudent() {
        $('#Name').val('');        
        $('#MobileNo').val('');        

        searchStudent();
    }
  
    function deleteStudent(item, e, id) {
        swal({
            title: 'Are you sure you want to delete?',
            type: 'warning',
            showCancelButton: true,
            showCloseButton: true,
            confirmButtonColor: '#3085d6',
            cancelButtonColor: '#d33',
            confirmButtonText: 'Yes'
        }).then((result) => {
            if (result.value) {
                $.ajax({
                    beforeSend: function (xhr) {
                        showLoader();
                    },
                    error: function (jqXHR, textStatus, errorThrown) {
                        ShowErrorMessage(jqXHR.status, errorThrown);
                    },
                    complete: function () {
                        hideLoader();
                    },
                    url: '/Student/Delete',
                    type: 'POST',
                    data: { id: id },
                    success: function (response) {
                        if (response.IsSuccess == true) {

                            var tr = $('#tblStudent  tbody tr#' + id)
                            var row = grdEmployee.row(tr);
                            var rowNode = row.node();
                            row.remove();
                            grdStudent.draw();

                            showToaster("Record deleted successfully", ToasterType.SUCCESS);
                        }
                        else {
                            showToaster(response.errorMessage, ToasterType.ERROR);
                        }
                    }
                });
            }
        });


    }
